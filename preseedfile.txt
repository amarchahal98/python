      boot-root ::                                            \

                      method{ format } format{ }              \

                      mountpoint{ /boot }                     \

              .                                               \
              100% 4096 100% linux-swap                       \
                      lv_name{ swap }                         \
                      method{ swap } format{ }                \
                      $lvmok{ }                               \
              .                                               \

              12288 12288 12288 ext4                          \
                      lv_name{ root }                         \
                      method{ lvm } format{ }                 \
                      use_filesystem{ } filesystem{ ext4 }    \
                      mountpoint{ / }                         \

              .                                               \
